Array sort

doc (fr)

let numbers8 = [3, 10, 2, 14, 7, 2, 9, 5];
let beatles = ["Ringo", "George", "Paul", "John"];

numbers8.sort(compareValues);
beatles.sort(compareValues);

function compareValues(a, b) {
    if (a < b) {
        // if a less than b
        return -1;
    } else if (a > b) {
        // if a greater than b
        return 1;
    } else {
        // a and b are equal
        return 0;
    }
}

console.log(numbers8);
console.log(beatles);

[ 2, 2, 3, 5, 7, 9, 10, 14 ]
[ 'George', 'John', 'Paul', 'Ringo' ]
