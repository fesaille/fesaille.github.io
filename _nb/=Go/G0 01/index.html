type Contact struct {
	phone, address string
}
type Employee struct {
	name    string
	salary  int
	contact Contact
}

func (c *Contact) changePhone(newPhone string) {
	c.phone = newPhone
}

e := Employee{
    name:   "Ross Geller",
    salary: 1200,
    contact: Contact{
        phone:   "011 8080 8080",
        address: "New Delhi, India",
    },
